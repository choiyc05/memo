backend/
└── src/
    └── utils/                # 외부 라이브러리 및 공통 모듈 관리
        ├── setting.py        # 환경 변수 관리 (Env, Config)

        ├── kafkasv.py        # Kafka 서비스 (Pd: 생산, Cs1: 이메일 발송, Cs2: AI 처리)

        ├── websc.py          # 실시간 로그 화면 출력 및 알람 처리 (WebSocket 등)
        ├── db.py             # 데이터베이스(DB) 연결 및 세션 관리
        ├── file.py           # 파일 업로드 및 다운로드 로직
        ├── tokenset.py       # 인증 토큰 관리 (Access Token, Refresh Token)
        └── crypto.py         # 데이터 암호화 및 복호화 처리
    └── apis/
        ├── user.py           # 회원 관리 (/put : 회원 가입, /patch : 회원 수정, /delete : 회원 탈퇴)
        ├── board.py          # /get : KPI 정보 조회 
        ├── auth.py           # 인증 관리 (/post : 로그인, /delete : 로그아웃)
        └── invite.py         # /put : 초대 관리
